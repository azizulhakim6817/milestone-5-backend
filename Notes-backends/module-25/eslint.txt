import tseslint from "typescript-eslint";

export default defineConfig({
  files: ["**/*.{js,ts}"],
  extends: [
    js.configs.recommended,

    tseslint.configs.strict,
    tseslint.configs.stylistic,
    {
      rules: {
        "no-console": "warn",
      },
    },
  ],
});
